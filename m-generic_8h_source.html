121/* Generate a sequence of the oplist that have been registered.
122
123 3 levels of indirection are needed:
124 * One level is the organisation.
125 * One level is the component of the organisation.
126 * One level is the oplist within the component of the organisation
127 Return the list of unevaluated oplist that are registered across all organizations, all components.
128 M*LIB Organisation (MLIB) is the default organisation and is always registered
129 Usage Domain: Nb organizations * Nb components * Nb oplist <= M_MAX_NB_ARGUMENT
130*/
131#define M_G3N_REGISTERED_ITEMS() \
132 M_CROSS_MAP( M_G3N_IS_PRESENT3, ( comp M_CROSS_MAP(M_G3N_IS_PRESENT2, (MLIB M_MAP(M_G3N_IS_PRESENT, M_G3N_SEQ_INT)), ( M_G3N_SEQ_INT ) ) ), ( M_G3N_SEQ_INT ) )

134#define M_G3N_IS_PRESENT(num) \
135 M_IF(M_PARENTHESIS_P(M_C(M_GENERIC_ORG_, num)() ))(M_DEFERRED_COMMA M_ID M_C(M_GENERIC_ORG_, num)(), )
136
137#define M_G3N_IS_PRESENT2(el1, num) \
138 M_IF(M_PARENTHESIS_P(M_C4(M_GENERIC_ORG_, el1, _COMP_, num)()))(M_DEFERRED_COMMA M_APPLY(M_C3, el1, _COMP_, M_ID M_C4(M_GENERIC_ORG_, el1, _COMP_, num)()), )

140#define M_G3N_IS_PRESENT3(el1, num) \
141 M_IF(M_OPLIST_P(M_C4(M_GENERIC_ORG_, el1, _OPLIST_, num)()))(M_DEFERRED_COMMA M_C4(M_GENERIC_ORG_, el1, _OPLIST_, num), )

269// for each item in the container
270// Same as M_EACH except it uses the generic macros
271#define M_G3N_EACHI(item, container, l_it, l_pass) \
272 (bool l_pass = true; l_pass; l_pass = false) \
273 for(sub_type(container) *item; l_pass ; l_pass = false) \
274 for(it_type(container) l_it; l_pass ; l_pass = false) \
275 for(it_first(l_it, container) ; \
276 !it_end_p(l_it) \
277 && (item = it_ref(l_it), true) ; \
278 it_next(l_it))
