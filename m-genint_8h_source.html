33/* GENINT is an internal container providing unique integers.
34 It has the following properties:
35 - it stores integer from [0..N) (N is fixed).
36 - an integer can have only one occurrence in the container.
37 - you can atomically push in / pop out integer from this container
38 provided that it is not already in the container.
39 - there are no order (like FIFO or stack)
40
41 This can be used to map integers to index of resources in a table.
42 At most we can support N = 32*64 = 2048 with the master limb usage.
43 For the typical usage of this container
44 (mapping hardware or software limited resources), this should be
45 enough.
46*/
47
48// Define the limb size used by genint
49typedef unsigned long long m_genint_limb_ct;
50
51/* Define a generator of unique integer (Lock Free) */
52typedef struct m_genint_s {
53 unsigned int n; // size of the container
54 unsigned int max; // number of allocated limb - 1
55 m_genint_limb_ct mask0; // mask of the last limb (constant)
56 m_genint_limb_ct mask_master; // mask of the master limb that controls others (constant)
57 atomic_ullong master; // master bitfield (which informs if a limb is full or not)
58 atomic_ullong *data; // the bitfield which informs if an integer is used or not
59} m_genint_t[1];
60
61// Define the max absolute supported value. It should be 2048 on most implementations.
62#define M_GENINT_MAX_ALLOC (M_GEN1NT_LIMBSIZE * (M_GEN1NT_LIMBSIZE - M_GEN1NT_ABA_CPT))
63
64// Define the size of a limb in bits.
65#define M_GEN1NT_LIMBSIZE ((unsigned)(sizeof(m_genint_limb_ct) * CHAR_BIT))
66
67// Define the contract of a genint
68#define M_GEN1NT_CONTRACT(s) do { \
69 M_ASSERT (s != NULL); \
70 M_ASSERT (s->n > 0 && s->n <= M_GENINT_MAX_ALLOC); \
71 M_ASSERT ((s->max+1) * M_GEN1NT_LIMBSIZE >= s->n); \
72 M_ASSERT (s->data != NULL); \
73 } while (0)
74
75// Define the limb one
76#define M_GEN1NT_ONE ((m_genint_limb_ct)1)
77
78#define M_GEN1NT_FULL_MASK ULLONG_MAX
79
80// Value returned in case of error (not integer available).
81#define M_GENINT_ERROR (UINT_MAX)
82
83/* 32 bits of the master mask are kept for handling the ABA problem.
84 * NOTE: May be too much. 16 bits should be more than enough. TBC
85 */
86#define M_GEN1NT_ABA_CPT 32
87#define M_GEN1NT_ABA_CPT_T uint32_t
88
89// Set the bit 'i' of the master limb, and increase ABA counter.
90#define M_GEN1NT_MASTER_SET(master, i) \
91 ((((master)& (~((M_GEN1NT_ONE<< M_GEN1NT_ABA_CPT)-1))) | (M_GEN1NT_ONE << (M_GEN1NT_LIMBSIZE - 1 - i))) \
92 |((M_GEN1NT_ABA_CPT_T)((master) + 1)))
93
94// Reset the bit i of the master limb, and increase ABA counter.
95#define M_GEN1NT_MASTER_RESET(master, i) \
96 (((master) & (~((M_GEN1NT_ONE<< M_GEN1NT_ABA_CPT)-1)) & ~(M_GEN1NT_ONE << (M_GEN1NT_LIMBSIZE - 1 - i))) \
97 |((M_GEN1NT_ABA_CPT_T)((master) + 1)))
98
99/* Initialize an integer generator (CONSTRUCTOR).
100 * Initially, the container is full of all the integers up to 'n-1'
101 * The typical sequence is to initialize the container, and pop
102 * the integer from it. Each pop integer is **unique** for all threads,
103 * meaning it can be used to index global unique resources shared
104 * for all threads.
105 */
106M_INLINE void
107m_genint_init(m_genint_t s, unsigned int n)
108{
109 M_ASSERT (s != NULL && n > 0 && n <= M_GENINT_MAX_ALLOC);
110 const size_t alloc = (n + M_GEN1NT_LIMBSIZE - 1) / M_GEN1NT_LIMBSIZE;
111 const unsigned int index = n % M_GEN1NT_LIMBSIZE;
112 atomic_ullong *ptr = M_MEMORY_REALLOC (atomic_ullong, NULL, alloc);
113 if (M_UNLIKELY_NOMEM (ptr == NULL)) {
114 M_MEMORY_FULL(alloc);
115 return;
116 }
117 s->n = n;
118 s->data = ptr;
119 s->max = (unsigned int) (alloc-1);
120 s->mask0 = (index == 0) ? M_GEN1NT_FULL_MASK : ~((M_GEN1NT_ONE<<(M_GEN1NT_LIMBSIZE-index))-1);
121 s->mask_master = (((M_GEN1NT_ONE << alloc) - 1) << (M_GEN1NT_LIMBSIZE-alloc)) >> M_GEN1NT_ABA_CPT;
122 atomic_init (&s->master, (m_genint_limb_ct)0);
123 for(unsigned int i = 0; i < alloc; i++)
124 atomic_init(&s->data[i], (m_genint_limb_ct)0);
125 M_GEN1NT_CONTRACT(s);
126}

128/* Clear an integer generator (Destructor) */
129M_INLINE void
130m_genint_clear(m_genint_t s)
131{
132 M_GEN1NT_CONTRACT(s);
133 M_MEMORY_FREE(s->data);
134 s->data = NULL;
135}

145/* Get an unique integer from the integer generator.
146 * NOTE: For a typical case, the amortized cost is one CAS per pop. */
147M_INLINE unsigned int
148m_genint_pop(m_genint_t s)
149{
150 M_GEN1NT_CONTRACT(s);
151 // First read master to see which limb is not full.
152 m_genint_limb_ct master = atomic_load(&s->master);
153 // While master is not full
154 while ((master >> M_GEN1NT_ABA_CPT) != s->mask_master) {
155 // Let's get the index i of the first not full limb according to master.
156 unsigned int i = m_core_clz64(~master);
157 M_ASSERT (i < M_GEN1NT_LIMBSIZE);
158 // Let's compute the mask of this limb representing the limb as being full
159 m_genint_limb_ct mask = s->mask0;
160 mask = (i == s->max) ? mask : M_GEN1NT_FULL_MASK;
161 unsigned int bit;
162 // Let's load this limb,
163 m_genint_limb_ct next, org = atomic_load(&s->data[i]);
164 do {
165 // If it is now full, we have been preempted by another.
166 if (M_UNLIKELY (org == mask))
167 goto next_element;
168 M_ASSERT (org != M_GEN1NT_FULL_MASK);
169 // At least one bit is free in the limb. Find one.
170 bit = M_GEN1NT_LIMBSIZE - 1 - m_core_clz64(~org);
171 M_ASSERT (bit < M_GEN1NT_LIMBSIZE);
172 M_ASSERT ((org & (M_GEN1NT_ONE<<bit)) == 0);
173 M_ASSERT (i * M_GEN1NT_LIMBSIZE + M_GEN1NT_LIMBSIZE - 1 - bit < s->n);
174 // Set the integer as being used.
175 next = org | (M_GEN1NT_ONE << bit);
176 // Try to reserve the integer
177 } while (!atomic_compare_exchange_weak (&s->data[i], &org, next));
178 // We have reserved the integer.
179 // If the limb is now full, try to update master
180 if (M_UNLIKELY(next == mask)) {
181 while (true) {
182 m_genint_limb_ct newMaster;
183 if (next == mask) {
184 newMaster = M_GEN1NT_MASTER_SET(master, i);
185 } else {
186 newMaster = M_GEN1NT_MASTER_RESET(master, i);
187 }
188 if (atomic_compare_exchange_weak (&s->master, &master, newMaster))
189 break;
190 // Fail to update. Reload limb to check if it is still full.
191 next = atomic_load(&s->data[i]);
192 }
193 }
194 // Return the new number
195 M_GEN1NT_CONTRACT(s);
196 return i * M_GEN1NT_LIMBSIZE + M_GEN1NT_LIMBSIZE - 1 - bit;
197 next_element:
198 // Reload master
199 master = atomic_load(&s->master);
200 }
201 M_GEN1NT_CONTRACT(s);
202 return M_GENINT_ERROR; // No more resource available
203}

205/* Restore a used integer in the integer generator.
206 * NOTE: For a typical case, the amortized cost is one CAS per pop */
207M_INLINE void
208m_genint_push(m_genint_t s, unsigned int n)
209{
210 M_GEN1NT_CONTRACT(s);
211 M_ASSERT (n < s->n);
212 const unsigned int i = n / M_GEN1NT_LIMBSIZE;
213 const unsigned int bit = M_GEN1NT_LIMBSIZE - 1 - (n % M_GEN1NT_LIMBSIZE);
214 m_genint_limb_ct master = atomic_load(&s->master);
215 // Load the limb
216 m_genint_limb_ct next, org = atomic_load(&s->data[i]);
217 do {
218 M_ASSERT ((org & (M_GEN1NT_ONE << bit)) != 0);
219 // Reset it
220 next = org & (~(M_GEN1NT_ONE << bit));
221 // Try to unreserve it.
222 } while (!atomic_compare_exchange_weak (&s->data[i], &org, next));
223 // if the limb was marked as full by master
224 m_genint_limb_ct mask = s->mask0;
225 mask = (i == s->max) ? mask : M_GEN1NT_FULL_MASK;
226 if (M_UNLIKELY (next != mask)) {
227 // Let's compute the mask of this limb representing the limb as being full
228 // Let's try to update master to say that this limb is not full
229 while (true) {
230 m_genint_limb_ct newMaster;
231 if (next == mask) {
232 newMaster = M_GEN1NT_MASTER_SET(master, i);
233 } else {
234 newMaster = M_GEN1NT_MASTER_RESET(master, i);
235 }
236 if (atomic_compare_exchange_weak (&s->master, &master, newMaster))
237 break;
238 // Fail to update. Reload limb to check if it is still full.
239 next = atomic_load(&s->data[i]);
240 }
241 }
242 M_GEN1NT_CONTRACT(s);
243}
